• Person(name, birth_date, phone_number)
  • Customer(phone_number, address)
    • FK: phone_number ⊆ Person(phone_number)
  • Address(phone_number, house_number, street, city)
    • FK: phone_number ⊆ Customer(phone_number)
  • Employee(phone_number, #_of_delivered_packages)
    • FK: phone_number ⊆ Person(phone_number)
  • IsSuperior(employee, superior)
    • FK: employee ⊆ Employee(phone_number)
    • FK: superior ⊆ Employee(phone_number)
  • Location(address)
  • Vehicle(address, licence_plate, employee, type)
    • FK: address ⊆ Location(address)
    • FK: employee ⊆ Employee(phone_number)
  • Warehouse(address, capacity, building_size)
    • FK: address ⊆ Location(address)
  • Package(deliver_address, timestamp, sender, weight)
    • FK: sender ⊆ Customer(phone_number)
  • IsIn(pkg_deliver_address, pkg_timestamp, pkg_sender, location)
    • FK: (pkg_deliver_address, pkg_timestamp, pkg_sender) ⊆ Package(deliver_address, timestamp, sender)
    • FK: location ⊆ Location(address)
  • Delivers(pkg_deliver_address, pkg_timestamp, pkg_sender, receiver, employee)
    • FK: (pkg_deliver_address, pkg_timestamp, pkg_sender) ⊆ Package(deliver_address, timestamp, sender)
    • FK: receiver ⊆ Customer(phone_number)
    • FK: employee ⊆ Employee(phone_number)
